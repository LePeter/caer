IF (NOT ENABLE_RECURRENTNET)
	SET(ENABLE_RECURRENTNET 0 CACHE BOOL "Enable the recurrent network module")
ENDIF()

IF (ENABLE_RECURRENTNET)
	IF(DYNAPSEFX2)
		SET(CAER_COMPILE_DEFINITIONS ${CAER_COMPILE_DEFINITIONS} -DENABLE_RECURRENTNET=1 PARENT_SCOPE)

		SET(CAER_HW_FILES modules/recurrentnet/recurrentnet.c)

		SET(CAER_C_SRC_FILES ${CAER_C_SRC_FILES} ${CAER_HW_FILES} PARENT_SCOPE)
	ELSE()
		MESSAGE("NOT COMPILING ENABLE_RECURRENTNET AS IT DEPENDS ON THE DYNAPSEFX2 MODULE")
	ENDIF()
ENDIF()
